4. Правки, улучшающие работу с константами:
	
	a - MAX_SPEED_LIMIT
	// новое название отражает смысл константы, установленную максимальную скорость транспорта
	
	day - HOURS_IN_DAY
	// акцент константы именно на "сутках", так как day - несколько размытое понятие
	
	'1' - ADD_STRING
	// присвоил константе значение, при котором производиться операция добавления
	
	'2' - DELETE_SYMBOL
	// присвоил константе значение, при котором производиться операция удаления
	
	'3' - GET_SYMBOL
	// присвоил константе значение, при котором происходит вывод символа
	
	'4' - MAKE_UNDO
	// присовил константе значение, при котором произволиться операция UNDO
	
	'5' - MAKE_REDO
	// присовил константе значение, при котором произволиться операция REDO
	// в предыдущих пяти примерах были внесены корректировки в задачу BastShoe, с курса "28 задач". Для удобства дальнейшего обслуживания и корректировки кода, константам присвоено значение всех реализованных операций.
	
	dist - BC_ROUTE_LENGTH
	// константа содержит информацию о длине велосипедного маршрута, который ужно проехать участникам заезда, привел константу в соответстующий вид
	
	while count <= 25 - while count <= NUM_TARGETS
	// 25 - это магическое число, нормативное число мишений, которые нужно поразить за один сет в стендовой стрельбе. Избавился от этого магического чисала, путем присвоения его значения константе
	
	func_call - MAX_FUNCTION_CALL
	// привел в соответствующий вид константу, которая содержит максимальное допустимое число обращений к функции
	
	workshops - NUM_OF_WORKSHOPS
	// привел в соответствующий вид константу, которая содержит количество цехов на предприятии
	
	gener - YEARLY_POWER_PLAN
	// привел в соответствующий вид константу, которая содержит информацию о годовом плане по выработке мощности на электростанции
